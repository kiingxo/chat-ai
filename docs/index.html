üìä Event System

ChatPilot dispatches custom events that you can listen to:

// Listen to ChatPilot events document.addEventListener('chatpilot:ready', (e) => { console.log('ChatPilot is ready!', e.detail.instance); }); document.addEventListener('chatpilot:open', (e) => { console.log('Chat opened', e.detail.instance); }); document.addEventListener('chatpilot:close', (e) => { console.log('Chat closed', e.detail.instance); }); document.addEventListener('chatpilot:error', (e) => { console.error('ChatPilot error:', e.detail.error); });
